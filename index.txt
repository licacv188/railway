function generateRandomName() {
  const chars = 'abcdefghijklmnopqrstuvwxyz';
  let result = '';
  for (let i = 0; i < 6; i++) {
    result += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return result;
}

function deleteNodes() {
  try {
    if (!UPLOAD_URL) return;
    if (!fs.existsSync(subPath)) return;

    let fileContent;
    try {
      fileContent = fs.readFileSync(subPath, 'utf-8');
    } catch {
      return null;
    }

    const decoded = Buffer.from(fileContent, 'base64').toString('utf-8');
    const nodes = decoded.split('\n').filter(line =>
      /(vless|vmess|trojan|hysteria2|tuic):\/\//.test(line)
    );

    if (nodes.length === 0) return;

    return axios.post(UPLOAD_URL + '/api/delete-nodes', JSON.stringify({ nodes }), {
      headers: { 'Content-Type': 'application/json' }
    }).catch(err => {
      return null;
    }), null;
  } catch (err) {
    return null;
  }
}

function getSystemArchitecture() {
  const arch = os.arch();
  return (arch === 'arm' || arch === 'arm64' || arch === 'aarch64') ? 'arm' : 'amd';
}

function downloadFile(filePath, url, callback) {
  const targetPath = filePath;

  if (!fs.existsSync(FILE_PATH)) {
    fs.mkdirSync(FILE_PATH, { recursive: true });
  }

  const writer = fs.createWriteStream(targetPath);

  axios({ method: 'get', url: url, responseType: 'stream' })
    .then(response => {
      response.data.pipe(writer);
      writer.on('finish', () => {
        writer.close();
        console.log('Download ' + path.basename(targetPath) + ' successfully');
        callback(null, targetPath);
      });
      writer.on('error', err => {
        fs.unlink(targetPath, () => {});
        const errMsg = 'Download ' + path.basename(targetPath) + ' failed: ' + err.message;
        console.error(errMsg);
        callback(errMsg);
      });
    })
    .catch(err => {
      const errMsg = 'Download ' + path.basename(targetPath) + ' failed: ' + err.message;
      console.error(errMsg);
      callback(errMsg);
    });
}

async function downloadFilesAndRun() {
  const arch = getSystemArchitecture();
  const files = getFilesForArchitecture(arch);

  if (files.length === 0) {
    console.log("Can't find a file for the current architecture");
    return;
  }

  const downloadPromises = files.map(file => {
    return new Promise((resolve, reject) => {
      downloadFile(file.fileName, file.fileUrl, (err, filePath) => {
        err ? reject(err) : resolve(filePath);
      });
    });
  });

  try {
    await Promise.all(downloadPromises);
  } catch (err) {
    console.error('Error downloading files:', err);
    return;
  }

  function setPermissions(files) {
    const permission = 0o775;
    files.forEach(file => {
      if (fs.existsSync(file)) {
        fs.chmod(file, permission, err => {
          if (err) {
            console.error('Empowerment failed for ' + file + ': ' + err);
          } else {
            console.log('Empowerment success for ' + file + ': ' + permission.toString(8));
          }
        });
      }
    });
  }

  const allPaths = [komariPath, webPath, botPath];
  setPermissions(allPaths);

  if (KOMARI_ENDPOINT && KOMARI_TOKEN) {
    const komariCmd = 'nohup ' + komariPath + ' -e ' + KOMARI_ENDPOINT + ' -t ' + KOMARI_TOKEN + ' >/dev/null 2>&1 &';
    try {
      await exec(komariCmd);
      console.log(komariName + ' is running');
      await new Promise(resolve => setTimeout(resolve, 1000));
    } catch (err) {
      console.error('komari-agent running error: ' + err);
    }
  } else {
    console.log('KOMARI variable is empty, skip running');
  }

  const webCmd = 'nohup ' + webPath + ' -c ' + FILE_PATH + '/config.json >/dev/null 2>&1 &';
  try {
    await exec(webCmd);
    console.log(webName + ' is running');
    await new Promise(resolve => setTimeout(resolve, 1000));
  } catch (err) {
    console.error('web running error: ' + err);
  }

  if (fs.existsSync(botPath)) {
    let argoArgs;

    if (ARGO_AUTH.match(/^[A-Z0-9a-z=]{120,250}$/)) {
      argoArgs = 'tunnel --edge-ip-version auto --no-autoupdate --protocol http2 run --token ' + ARGO_AUTH;
    } else if (ARGO_AUTH.match(/TunnelSecret/)) {
      argoArgs = 'tunnel --edge-ip-version auto --config ' + FILE_PATH + '/tunnel.yml run';
    } else {
      argoArgs = 'tunnel --edge-ip-version auto --no-autoupdate --protocol http2 --logfile ' + FILE_PATH + '/boot.log --loglevel info --url http://localhost:' + ARGO_PORT;
    }

    try {
      await exec('nohup ' + botPath + ' ' + argoArgs + ' >/dev/null 2>&1 &');
      console.log(botName + ' is running');
      await new Promise(resolve => setTimeout(resolve, 2000));
    } catch (err) {
      console.error('Error executing command: ' + err);
    }
  }

  await new Promise(resolve => setTimeout(resolve, 5000));
}

function getFilesForArchitecture(arch) {
  let files;

  if (arch === 'arm') {
    files = [
      { fileName: webPath, fileUrl: 'https://arm64.ssss.nyc.mn/web' },
      { fileName: botPath, fileUrl: 'https://arm64.ssss.nyc.mn/bot' }
    ];
  } else {
    files = [
      { fileName: webPath, fileUrl: 'https://amd64.ssss.nyc.mn/web' },
      { fileName: botPath, fileUrl: 'https://amd64.ssss.nyc.mn/bot' }
    ];
  }

  if (KOMARI_ENDPOINT && KOMARI_TOKEN) {
    const komariUrl = arch === 'arm'
      ? 'https://github.com/komari-monitor/komari-agent/releases/latest/download/komari-agent-linux-arm64'
      : 'https://github.com/komari-monitor/komari-agent/releases/latest/download/komari-agent-linux-amd64';
    files.unshift({ fileName: komariPath, fileUrl: komariUrl });
  }

  return files;
}

function argoType() {
  if (!ARGO_AUTH || !ARGO_DOMAIN) {
    console.log('ARGO_DOMAIN or ARGO_AUTH variable is empty, use quick tunnels');
    return;
  }

  if (ARGO_AUTH.includes('TunnelSecret')) {
    fs.writeFileSync(path.join(FILE_PATH, 'tunnel.json'), ARGO_AUTH);

    const tunnelId = ARGO_AUTH.split('"')[11];
    const yml = `
  tunnel: ${tunnelId}
  credentials-file: ${path.join(FILE_PATH, 'tunnel.json')}
  protocol: http2
  
  ingress:
    - hostname: ${ARGO_DOMAIN}
      service: http://localhost:${ARGO_PORT}
      originRequest:
        noTLSVerify: true
    - service: http_status:404
  `;
    fs.writeFileSync(path.join(FILE_PATH, 'tunnel.yml'), yml);
  } else {
    console.log('ARGO_AUTH mismatch TunnelSecret,use token connect to tunnel');
  }
}

async function extractDomains() {

  } else {
    try {
      const logContent = fs.readFileSync(path.join(FILE_PATH, 'boot.log'), 'utf-8');
      const lines = logContent.split('\n');
      const domains = [];

      lines.forEach(line => {
        const match = line.match(/https?:\/\/([^ ]*trycloudflare\.com)\/?/);
        if (match) {
          const domain = match[1];
          domains.push(domain);
        }
      });

      if (domains.length > 0) {
        argoDomain = domains[0];
        console.log('ArgoDomain:', argoDomain);

        console.log('ArgoDomain not found, re-running bot to obtain ArgoDomain');

        async function killBot() {
          try {
            if (process.platform === 'win32') {
              await exec('taskkill /f /im ' + botName + '.exe > nul 2>&1');
            } else {
              await exec('pkill -f "[' + botName.charAt(0) + ']' + botName.substring(1) + '" > /dev/null 2>&1');

          await extractDomains();

          console.error('Error executing command: ' + err);
        }
      }
    } catch (err) {
      console.error('Error reading boot.log:', err);

  async function getLocationInfo() {

async function uploadNodes() {
  if (UPLOAD_URL && PROJECT_URL) {
    const subUrl = PROJECT_URL + '/' + SUB_PATH;
    const payload = { subscription: [subUrl] };

    try {
      const response = await axios.post(UPLOAD_URL + '/api/add-subscriptions', payload, {
        headers: { 'Content-Type': 'application/json' }
      });
      if (response && response.status === 200) {
        console.log('Subscription uploaded successfully');
        return response;
      }
      return null;
    } catch (err) {
      if (err.response) {
        if (err.response.status === 400) {
        }
      }
    }
  } else if (UPLOAD_URL) {
    if (!fs.existsSync(listPath)) return;

    const fileContent = fs.readFileSync(listPath, 'utf-8');
    const nodes = fileContent.split('\n').filter(line =>
      /(vless|vmess|trojan|hysteria2|tuic):\/\//.test(line)
    );

    if (nodes.length === 0) return;

    const payload = JSON.stringify({ nodes });

    try {
      const response = await axios.post(UPLOAD_URL + '/api/add-nodes', payload, {
        headers: { 'Content-Type': 'application/json' }
      });
      if (response && response.status === 200) {
        console.log('Nodes uploaded successfully');
        return response;
      }
      return null;
    } catch (err) {
      return null;
    }
  } else {
    return;
  }
}
